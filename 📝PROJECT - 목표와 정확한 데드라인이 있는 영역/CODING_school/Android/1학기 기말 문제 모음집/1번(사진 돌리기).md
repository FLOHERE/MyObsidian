- 1번 java
```java
{
   Button btn_1, btn_plus;
   EditText edit_text_Url, edt_1, edt_2;

   @Override
   protected void onCreate(Bundle savedInstanceState) {
       super.onCreate(savedInstanceState);
       EdgeToEdge.enable(this);
       setContentView(R.layout.activity_main);
       
     btn_1 = findViewById(R.id.btn_1);
     edit_text_Url = findViewById(R.id.edit_text_Url);
     edt_1 = findViewById(R.id.edit_1);
     edt_2 = findViewById(R.id.edit_2);
     btn_plus = findViewById(R.id.btn_plus);

   //sub1에 데한 url 데이터 전달 이벤트 버튼
     btn_1.setOnClickListener(new View.OnClickListener() {
         @Override
         public void onClick(View v) {
             String siteUrl = edit_text_Url.getText().toString();
             Intent sub1 = new Intent(MainActivity.this, sub1.class);
             sub1.putExtra("siteUrl", siteUrl);
             //사이트 url에 대한 데이터 값 전달
             //m.naver.com, m.nate.com, m.daum.net
             //putExtr 메서드는 1. 안드로이드에서 intent 객체에 데이터를 첨부하는데 사용
             //2. intent에 여러 유형으로 사용할 수 있다
             //3. putExtra메서드는 여러 유형의 데이터를 전달 할 수 있다.
             startActivity(sub1);
         }
     });
     //+버튼의 데이터 전달 이벤트
   btn_plus.setOnClickListener(new View.OnClickListener() {
       @Override
       public void onClick(View v) {
           Intent intent = new Intent(getApplicationContext(), sub2.class);
           //새로운 intent 객체를 생성하면서 현재 etApplicationContext에서
           // sub2.class라는 곳으로 이동할 것임을 명시함(명시적 인텐트)
           intent.putExtra("edt_1", Integer.parseInt(edt_1.getText().toString()));
          // intent.putExtra 이 코드는 Intent에 추가 데이터를 넣는 역할을 함
           intent.putExtra("edt_2", Integer.parseInt(edt_2.getText().toString()));
           // intent.putExtra 이 코드는 Intent에 추가 데이터를 넣는 역할을 함
           startActivityForResult(intent, 0);
           //
       }
	});
   }
   @Override
   protected void onActivityResult(int requestCode, int resultCode, Intent data) {
       super.onActivityResult(requestCode, resultCode, data);
       //액티비티가 main이 있고 sub가 있으면,
       //main 액티비티에서 sub 액티비티를 호출하여 넘어갔다가 다시 
       //main액티비티로 돌아올 때 사용되는 기본 메서드
       if (resultCode == RESULT_OK) {
           int data_back = data.getIntExtra("data_back", 0);
           Toast.makeText(getApplicationContext(), "합계 출력 : " + data_back, Tost.LENGTH_SHORT).show();
    }
   }
}
```
